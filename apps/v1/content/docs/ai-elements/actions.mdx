## Usage

```tsx
import { Actions, Action } from "@/ai-elements/actions"
```

```tsx
<Actions>
  <Action tooltip="Copy">
    <CopyIcon className="size-4" />
  </Action>
  <Action tooltip="Like">
    <ThumbsUpIcon className="size-4" />
  </Action>
</Actions>
```

## Examples

### With Handlers

```tsx
<Actions>
  <Action tooltip="Copy" onPress={() => copyToClipboard()}>
    <CopyIcon className="size-4" />
  </Action>
  <Action tooltip="Regenerate" onPress={() => regenerate()}>
    <RefreshCwIcon className="size-4" />
  </Action>
</Actions>
```

### Without Tooltips

```tsx
<Actions>
  <Action label="Copy">
    <CopyIcon className="size-4" />
  </Action>
</Actions>
```

## API Reference

### Actions

Container for action buttons.

| Prop | Type | Default | Description |
|---|---|---|----|
| `className` | `string` | - | Additional CSS classes |

### Action

| Prop | Type | Default | Description |
|---|---|---|----|
| `tooltip` | `string` | - | Tooltip text (shows on hover) |
| `label` | `string` | - | Accessibility label |
| `variant` | `"ghost" \| "outline" \| ...` | `"ghost"` | Button variant |
| `size` | `"sm" \| "default" \| ...` | `"sm"` | Button size |
